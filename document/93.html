json与xml

Json操作使用 json类；XML操作使用 xml类；Json 与 XML 所操作的数据必须为UTF-8编码，转码请使用 util::convert()

数组 => XML


	$arr = [
		'a' => 1,
		'b' => 2
	];
	$str = boa::xml()->write($arr);

XML => 数组


	$str = '<?xml version="1.0" encoding="UTF-8"?>
<boa>
	<a>1</a>
	<b>2</b>
</boa>';
	$arr = boa::xml()->read($str);

数组 => JSON


	$arr = [
		'a' => 1,
		'b' => 2
	];
	$str = boa::json()->encode($arr);

JSON => 数组


	$str = '{"a":1,"b":2}';
	$arr = boa::json()->decode($str);
