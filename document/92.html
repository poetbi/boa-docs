图像处理

格式转换


	$im = boa::image();
	$im->open(BS_WWW .'file/1.webp');
	$im->save(BS_WWW .'file/1.png');

水印


	$im = boa::image();
	$im->cfg('wm_type', 1);
	for(...){ // 批量加文字水印
		$im->open(BS_WWW ."res/$i.png");
		$im->watermark();
		$im->save(BS_WWW ."file/$i.png");
	}

缩略图


	$im = boa::image();
	for(...){ // 批量生成缩略图
		$im->open(BS_WWW ."res/$i.png");
		$im->thumbnail();
		$im->save(BS_WWW ."thumb/$i.png");
	}

默认配置参见 图像配置，更多功能参见 image类
