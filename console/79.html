命令行配置

配置项默认值类型说明
DOCUMENT_ROOTBS_WWWstring同$_SERVER['DOCUMENT_ROOT']
SERVER_PORT80int同$_SERVER['SERVER_PORT']
SERVER_NAMElocalhoststring同$_SERVER['SERVER_NAME']
HTTP_HOSTlocalhoststring同$_SERVER['HTTP_HOST']
REMOTE_ADDR127.0.0.1string同$_SERVER['REMOTE_ADDR']
REMOTE_PORT8888int同$_SERVER['REMOTE_PORT']
REQUEST_METHODGETstring同$_SERVER['REQUEST_METHOD']
REQUEST_SCHEMEhttpstring同$_SERVER['REQUEST_SCHEME']
REQUEST_URIstring同$_SERVER['REQUEST_URI']，默认由程序算出
QUERY_STRINGstring同$_SERVER['QUERY_STRING']，默认由程序算出
PATH_INFOstring同$_SERVER['PATH_INFO']，默认由程序算出
HTTP_COOKIEstring同$_SERVER['HTTP_COOKIE']，如：PHPSESSID=xxx
HTTP_USER_AGENTConsole boaPHP/5.5string同$_SERVER['HTTP_USER_AGENT']
