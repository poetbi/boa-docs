$stmt = $db->stmt_query('CALL sp_test(?, ?, ?)');
$param = [
	'in',
	[10, PDO::PARAM_INT],
	['@out', PDO::PARAM_STR | PDO::PARAM_INPUT_OUTPUT, 4000]
];
$stmt->execute($param);
$arr = $stmt->one();
